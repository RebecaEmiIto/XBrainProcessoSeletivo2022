CREATE TABLE Vendedores(
	VendedorID INT NOT NULL AUTO_INCREMENT,
	DataInsercao DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
	VendedorNome VARCHAR(255) NOT NULL,
	PRIMARY KEY (VendedorID)
);

CREATE TABLE Vendas(
	VendaID INT NOT NULL AUTO_INCREMENT,
	DataVenda DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
	Valor VARCHAR(100) NOT NULL,
	VendedorID INT NOT NULL,
	VendedorNome VARCHAR(255),
	PRIMARY KEY (VendaID),
	CONSTRAINT FK_Vendedor FOREIGN KEY (VendedorID) REFERENCES Vendedores(VendedorID)
);
